<script>
  const WEBHOOK_RAW =
    'https://n8n-production-1ddf.up.railway.app/webhook/chat-search';

  const CHECK_RAW =
    'https://n8n-production-1ddf.up.railway.app/webhook/check-access';

  const PROXY = 'https://corsproxy.io/?';

  let IS_SUBSCRIBED = false;

  async function checkAccess() {

    const phone = document
      .getElementById('phone-input')
      .value
      .trim()
      .replace(/\D/g, '')
      .slice(-10);

    if (!phone || phone.length < 10) {
      alert('Please enter a valid 10-digit phone number');
      return;
    }

    const badge = document.getElementById('access-badge');

    badge.textContent = 'Checking...';

    try {

      const res = await fetch(
        PROXY + encodeURIComponent(CHECK_RAW),
        {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({ phone })
        }
      );

      const data = await res.json();

      if (data.is_subscribed) {

        IS_SUBSCRIBED = true;

        badge.textContent = '✅ Pro Access Active';
        badge.style.color = '#2e7d32';

        document
          .getElementById('payment-section')
          .classList.remove('visible');

      } else {

        IS_SUBSCRIBED = false;

        badge.textContent = '🔒 Free Access';
        badge.style.color = '#e65100';
      }

    } catch (e) {

      badge.textContent = 'Error checking';
      badge.style.color = '#c62828';

      console.error(e);
    }
  }

  function showPayment() {

    const section = document.getElementById('payment-section');

    section.classList.add('visible');

    section.scrollIntoView({
      behavior: 'smooth'
    });

    section.style.border = '2px solid #e65100';
  }

  function fillChat(text) {

    document.getElementById('chat-input').value = text;

    sendMessage();
  }

  async function sendMessage() {

    const input = document.getElementById('chat-input');

    const btn = document.getElementById('send-btn');

    const messages = document.getElementById('messages');

    const text = input.value.trim();

    if (!text) return;

    btn.disabled = true;

    input.value = '';

    messages.innerHTML += `
      <div class="msg-user">${text}</div>
    `;

    const typingEl = document.createElement('div');

    typingEl.className = 'typing-indicator';

    typingEl.id = 'typing';

    typingEl.innerHTML = `
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
    `;

    messages.appendChild(typingEl);

    messages.scrollTop = messages.scrollHeight;

    try {

      const res = await fetch(
        PROXY + encodeURIComponent(WEBHOOK_RAW),
        {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            message: text,
            is_subscribed: IS_SUBSCRIBED
          })
        }
      );

      const data = await res.json();

      document.getElementById('typing')?.remove();

      messages.innerHTML += `
        <div class="msg-ai">
          ${data.reply || 'Here are your results'}
        </div>
      `;

      if (data.candidates && data.candidates.length > 0) {

        let cardsHTML = '<div class="cards-grid">';

        data.candidates.forEach(c => {

          const name = IS_SUBSCRIBED
            ? (c.Full_Name || 'Candidate')
            : (c.Masked_Name || 'Candidate');

          cardsHTML += `
            <div class="candidate-card">

              <div class="${IS_SUBSCRIBED ? 'pro-badge' : 'free-badge'}">
                ${IS_SUBSCRIBED ? '✅ Pro View' : 'Free Preview'}
              </div>

              <div class="card-name">
                ${name}
              </div>

              <div class="card-role">
                ${c.Current_Role || ''} · ${c.City || ''}
              </div>

              <div class="card-score">
                ${c.AI_Score || '—'} AI Score
              </div>

              <div class="card-detail">
                📅 ${c.Total_Exp_Years || '—'} yrs exp
              </div>

              <div class="card-detail">
                💰 ₹${c.Expected_Salary_LPA || '—'} LPA expected
              </div>

              <div class="card-detail">
                ⏱ ${c.Notice_Period_Days || '—'} days notice
              </div>

              ${
                IS_SUBSCRIBED
                  ? `
                    <div class="card-detail">
                      📞 ${c.Phone || '—'}
                    </div>

                    <div class="card-detail">
                      📧 ${c.Email || '—'}
                    </div>
                  `
                  : ''
              }

              <div class="card-skills">
                ${c.Skills || ''}
              </div>

              <button
                class="unlock-btn"
                onclick="${
                  IS_SUBSCRIBED
                    ? ''
                    : 'showPayment()'
                }"
              >
                ${
                  IS_SUBSCRIBED
                    ? '📋 Full Profile Unlocked'
                    : '💳 Unlock — ₹999 ↓'
                }
              </button>

            </div>
          `;
        });

        cardsHTML += '</div>';

        messages.innerHTML += cardsHTML;
      }

      messages.scrollTop = messages.scrollHeight;

    } catch (err) {

      document.getElementById('typing')?.remove();

      messages.innerHTML += `
        <div class="msg-ai" style="color:#c62828;">
          Something went wrong. Please try again.
        </div>
      `;

      console.error(err);
    }

    btn.disabled = false;

    input.focus();
  }

  document
    .getElementById('chat-input')
    .addEventListener('keydown', e => {

      if (e.key === 'Enter') {
        sendMessage();
      }

    });
</script>
